beta=0                       #pi * beta/2 - angle 
m=beta/(2-beta)

def pde(eta, out):
    u,tau=out[:,0:1],out[:,1:2]
    u_eta = dde.grad.jacobian(out, eta, i=0, j=0)
    tau_eta= dde.grad.jacobian(out, eta, i=1, j=0)
    
    term1=tau_eta/u_eta
    term1=dde.grad.jacobian(term1, eta)

    ode_1=(1/u_eta)*term1 - beta*(1-u*u)*(1/(u_eta*tau*tau))*tau_eta # + (1-2*beta)*u/tau 
    
    ode_2=tau-u_eta

    return [ode_1,ode_2]

eps=0.1
geom = dde.geometry.Interval(0,5)

def boundary_b(eta,on_boundary):
    return on_boundary and np.isclose(eta[0], 0)
def boundary_t(eta,on_boundary):
    return on_boundary and np.isclose(eta[0], 5)

bc_1 = dde.DirichletBC(geom, lambda X: 0, boundary_b, component=0)
bc_2 = dde.DirichletBC(geom, lambda X: 1, boundary_t, component=0)
bc_3 = dde.DirichletBC(geom, lambda X: eps, boundary_t, component=1)
#bc_4 = dde.OperatorBC(geom, lambda X, y: dde.grad.hessian(y,X) - 0, boundary_r)
#bc_4 = dde.OperatorBC(geom, hessian_r, boundary_r)

bc = [bc_1, bc_2, bc_3]

data = dde.data.PDE(geom, pde, bc, num_domain=60, num_boundary= 2, num_test=40)
